== Sensitive Code Example

----
fun setup(activity: MainActivity, br: BroadcastReceiver, filter: IntentFilter) {
    activity.registerReceiver(br, filter) // Sensitive
}
----

== Compliant Solution

----
fun setup(activity: MainActivity, br: BroadcastReceiver, filter: IntentFilter) {
    activity.registerReceiver(br, filter, Manifest.permission.SEND_SMS, null)
}
----
